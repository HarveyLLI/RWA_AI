// Antalpha RWA Yield Skills
import { AntalphaCLI } from '@agentwealth/cli';

const client = new AntalphaCLI({
  apiKey: process.env.ANTALPHA_API_KEY,
  network: 'ethereum'
});

// Query available products
const products = await client.listProducts();
// → [{ id: "btc-usdt-stable", apy: 6.0, lock: "6m" }]

// Subscribe to yield product
const tx = await client.subscribe({
  productId: 'btc-usdt-stable',
  amount: 1000,          // 1000 USDT
  collateral: 'BTC'      // Over-collateralized
});

console.log(`✓ Subscribed! TX: ${tx.hash}`);
// ✓ Subscribed! TX: 0x3a7f...8b2c
